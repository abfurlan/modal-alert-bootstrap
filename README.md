<h1>Alerta com modal bootstrap</h1>
Exibir alertas com o componente modal Bootstrap.

<h2>Classes</h2>
<ul>
  <li>primary</li>
  <li>success</li>
  <li>danger</li>
  <li>warning</li>
  <li>info</li>
  <li>default</li>
</ul>
<h2>Efeitos</h2>
<ul>
  <li>fade</li>
  <li>zoom</li>
  <li>newspaper</li>
  <li>move-horizontal</li>
  <li>move-from-top</li>
  <li>zoom-out</li>
</ul>
<h2>Exemplo</h2>
<pre>
  <code>
    $.alert({
              class:"primary",
              title:'Título',
              message:'Sua mensagem aqui...',
              effect:"zoom"
            });
  </code>
</pre>
